Powershell - most efficient way to delete first X lines from a file.

$infile = "$env:TEMP\test.txt"
$outfile = "$env:TEMP\test2.txt"

# fill infile with goop
1..100 | % { "Line $_ : Moo" } | Out-File $infile

$skiplines = 5

$reader = New-Object System.IO.StreamReader $infile
$writer = New-Object System.IO.StreamWriter $outfile

for($i = 0; $i -lt $skiplines; $i++) {
    $null = $reader.ReadLine()
}

while(-not $reader.EndOfStream) {
    $writer.WriteLine($reader.ReadLine())
}

$reader.Close()
$writer.Close()
